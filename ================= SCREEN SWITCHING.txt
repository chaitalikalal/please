/* ================= SCREEN SWITCHING ================= */
function goTo(id) {
  document.querySelectorAll(".screen").forEach(screen => screen.classList.remove("active"));
  document.getElementById(id).classList.add("active");
}

/* ================= MESSAGE SHARING ================= */
function generateLink() {
  const msg = document.getElementById("userMessage").value.trim();
  if (!msg) {
    alert("Please write a message first ðŸ’Œ");
    return;
  }
  const encoded = encodeURIComponent(msg);
  const url = `${window.location.origin}${window.location.pathname}?msg=${encoded}`;
  document.getElementById("shareLink").innerHTML =
    `Share this link with your friend:<br><a href="${url}">${url}</a>`;
}

/* ================= MEMORY GAME ================= */
const memoryData = ["ðŸŽ","ðŸŒ","ðŸ“","ðŸ‡","ðŸ’","ðŸ"];
let memoryCards = [];
let firstCard = null;
let lockBoard = false;

function startMemoryGame() {
  memoryCards = [...memoryData, ...memoryData]; // duplicate for pairs
  memoryCards.sort(() => 0.5 - Math.random()); // shuffle

  const puzzleContainer = document.querySelector(".puzzle-box");
  puzzleContainer.innerHTML = "";

  memoryCards.forEach(symbol => {
    const card = document.createElement("div");
    card.classList.add("memory-card");
    card.dataset.symbol = symbol;
    card.innerHTML = `<div class="front">?</div><div class="back">${symbol}</div>`;
    card.addEventListener("click", flipCard);
    puzzleContainer.appendChild(card);
  });
}

function flipCard() {
  if (lockBoard || this.classList.contains("flipped")) return;

  this.classList.add("flipped");

  if (!firstCard) {
    firstCard = this;
    return;
  }

  const secondCard = this;
  if (firstCard.dataset.symbol === secondCard.dataset.symbol) {
    firstCard = null;
    checkWin();
  } else {
    lockBoard = true;
    setTimeout(() => {
      firstCard.classList.remove("flipped");
      secondCard.classList.remove("flipped");
      firstCard = null;
      lockBoard = false;
    }, 800);
  }
}

function checkWin() {
  const flippedCards = document.querySelectorAll(".memory-card.flipped");
  if (flippedCards.length === memoryCards.length) {
    alert("ðŸŽ‰ You matched all cards! Great job!");
  }
}
